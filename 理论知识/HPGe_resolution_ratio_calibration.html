  //

      A[i] = h_gammaraw->GetBinContent(h_gammaraw->FindBin(e[i]));
      ffit[i] = new TF1(Form("ffit%d", i), fun, 0, 2500, 5);// 0  2500   5 
      ffit[i]->SetParameter(0, A[i]); //par[0]-
      ffit[i]->SetParameter(1, e[i]);   //par[1]-
      ffit[i]->SetParameter(2, 1.0);    //par[2]-sigma
      ffit[i]->SetParameter(3, 1.0e3);  //par[3]-
      ffit[i]->SetParameter(4, 0.01);   //par[4]-

      fr = h_gammaraw->Fit(ffit[i], "SQ+", "", e[i] - 5, e[i] + 5);
      sigma[i] = 2.35*ffit[i]->GetParameter(2);

  }

用高斯拟合看看两个sigma的差异¶

In [6]:
TF1 *fg[5];
double Ag[5];
double sigmag[5];
double parg[5][3];

for(int i=0; i<5; i++) {
    Ag[i] = h_gammaraw->GetBinContent(h_gammaraw->FindBin(e[i]));
    fg[i] = new TF1(Form("fg%d", i), "gaus");
    fg[i]->SetParameters(Ag[i], e[i], 1);//constant,mean, sigma
    fr = h_gammaraw->Fit(fg[i], "SQ+", "", e[i]-5, e[i]+5);//superimpose TF1 to hx1
    parg[i][0] = fg[i]->GetParameter(1);
    parg[i][1] = fg[i]->GetParameter(2);
    parg[i][2] = fr->Chi2()/fr->Ndf();
    sigmag[i] = 2.35*parg[i][1];
    cout << Form("peaks = %4.1f, sigma = %.2f, chi2/ndf = %.2f, sigmag = %.2f",
                 parg[i][0], parg[i][1], parg[i][2], sigmag[i]) << endl;
}
c1->Draw();
